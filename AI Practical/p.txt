public class Invoice_Control {
    public String message { get; set; }
    public Decimal amount { get; set; }
    public String customerEmail { get; set; }
    public List<Invoice__c> customerInvoices { get; set; }

    public void createInvoice() {
        customerInvoices = new List<Invoice__c>(); // Reset the invoice list

        if (amount <= 0 || String.isEmpty(customerEmail)) {
            message = 'Error: Amount must be greater than 0 and email is required.';
            return;
        }

        try {
            // Create the invoice
            Invoice__c inv = new Invoice__c(
                Amount__c = amount, 
                Customer_Email__c = customerEmail, 
                Due_Date__c = Date.today().addDays(30),
                Name = 'INV-' + String.valueOf(Datetime.now().getTime()) // simple unique name
            );
            insert inv;

            // Re-query the record to ensure we get the Name
            inv = [SELECT Name FROM Invoice__c WHERE Id = :inv.Id LIMIT 1];

            // Set success message with name
            message = 'Invoice Created: ' + inv.Name;

            // Query all customer invoices
            customerInvoices = [SELECT Name, Amount__c, Due_Date__c FROM Invoice__c WHERE Customer_Email__c = :customerEmail];

        } catch (Exception e) {
            message = 'Error: ' + e.getMessage();
        }
    }
}


<apex:page controller="Invoice_Control">
    <apex:form>
        <apex:pageMessages />

        <label>Amount:</label>
        <apex:inputText value="{!amount}" /><br/>

        <label>Customer Email:</label>
        <apex:inputText value="{!customerEmail}" /><br/>

        <apex:commandButton value="Create Invoice" action="{!createInvoice}" rerender="msg,invoiceList"/>

        <apex:outputPanel id="msg">
            <apex:outputText value="{!message}" style="color:green;font-weight:bold;" />
        </apex:outputPanel>

        <apex:outputPanel id="invoiceList" style="margin-top: 20px;">
            <apex:outputText rendered="{!ISNULL(customerInvoices) || customerInvoices.size == 0}" value="No invoices found for this customer." />
            
            <apex:dataTable value="{!customerInvoices}" var="inv" rendered="{!NOT(ISNULL(customerInvoices)) && customerInvoices.size > 0}"
                border="1" cellpadding="5" cellspacing="0" style="width: 100%; text-align: center;">
    			<apex:column headerValue="Invoice Name">
        			<apex:outputText value="{!inv.Name}" />
    			</apex:column>
    			<apex:column headerValue="Amount">
        			<apex:outputText value="{!inv.Amount__c}" />
    			</apex:column>
    			<apex:column headerValue="Due Date">
        			<apex:outputText value="{!inv.Due_Date__c}" />
    			</apex:column>
</apex:dataTable>

        </apex:outputPanel>
    </apex:form>
</apex:page>
